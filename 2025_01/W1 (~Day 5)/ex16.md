```ruby
#include <string>
#include <vector>

using namespace std;

vector<long long> solution(int x, int n) {
    vector<long long> answer;
    
    answer.push_back(x);
    
    if((2<=n)&&(n<=1000)){
        int temp = x;
        for(int i=1; i<n; i++){
            x += temp;
            
            answer.push_back(x);
        }
    }
    
    return answer;
}
```
