#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int n) {
    int answer = 0;
    int sum = 0;
    
    if((0< n) && (n <= 1000)){
        for(int i=0; i<=n; i++){
            if(i%2==0){
                sum+=i;
            }
        }
        answer = sum;
    }
    return answer;
}
