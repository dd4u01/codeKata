```ruby
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int n) {
    int answer = 0;
    
    if((3<=n)&&(n<=1000000)){
        for(int x=1; x<n; x++){
            if(n % x == 1){
                answer = x;
                
                break;
            }
        }
    }
    
    return answer;
}
```
