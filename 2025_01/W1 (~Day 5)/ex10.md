```ruby
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

// numbers_len은 배열 numbers의 길이입니다.
double solution(int numbers[], size_t numbers_len) {
    double answer = 0;
    double sum = 0;
    double avg = 0;
    
    if((1<=numbers_len)&&(numbers_len<=100)){
        for(int i=0; i<numbers_len; i++){
            if((0<=numbers[i])&&(numbers[i]<=1000)){
                sum += numbers[i];
            }
        }
        avg = sum / numbers_len;
        answer = avg;
    }
    
    return answer;
}
```
