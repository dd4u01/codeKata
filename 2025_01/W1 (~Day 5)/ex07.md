#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int num1, int num2) {
    
    int answer=0;
    
    if((0<num1 && num1<=100)&&(0<num2 && num2<=100)){
        
        // 1. 소수점 표기를 위해 num1 혹은 num2 를 double 형으로 변환해준다.
        // 2. 계산한 값을 다시 정수 int형으로 변환해준다.
        answer = (int)(((double)num1 / num2) * 1000);
    }

    return answer;
}
